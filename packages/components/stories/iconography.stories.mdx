import { Meta, IconGallery, IconItem } from '@storybook/addon-docs'
import { Icon } from '../src'
import * as phosphorIcons from 'phosphor-react'

<Meta title="Styles/Iconography" />

export function AllIcons({ icons: allIcons }) {
  return allIcons.map((iconName) => (
    <IconItem name={iconName} key={iconName}>
      <Icon name={iconName} />
    </IconItem>
  ))
}

# Iconography

OpenCRVS UI-Kit is uses the open source [Feather icons](https://github.com/feathericons/feather) library with some additonal custom icons. Visit the [&lt;Icon&gt component](/docs/data-icon--default) story page for more information.
